31 {
32  typename MAPTYPE::const_iterator iter = map_.find(key);
33  if (!(iter->second == data))
34  {
35  STXXL_VERBOSE2("iter=(" << (*iter).first << ":" << (*iter).second << ")");
36  STXXL_VERBOSE2("key=" << key);
37  STXXL_VERBOSE2("data=" << data);

39  }

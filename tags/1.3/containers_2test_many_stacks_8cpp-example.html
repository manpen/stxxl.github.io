containers/test_many_stacks.cpp

This is an example of how to use stxxl::STACK_GENERATOR class to generate an external stack type with stxxl::grow_shrink_stack implementation, four blocks per page, block size 4096 bytes

/***************************************************************************

* (See accompanying file LICENSE_1_0.txt or copy at
* http://www.boost.org/LICENSE_1_0.txt)
**************************************************************************/
#include <stxxl/stack>
int main(int argc, char * argv[])
{
typedef stxxl::STACK_GENERATOR<int, stxxl::external, stxxl::normal, 4, 4096>::result ext_stack_type;
if (argc < 2)
{
STXXL_MSG("Usage: " << argv[0] << " number_of_stacks");
return -1;
}
ext_stack_type * my_stacks = new ext_stack_type[atoi(argv[1])];
delete[] my_stacks;
return 0;
}

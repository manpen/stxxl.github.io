34 //! Buffered input stream, reading the items in the blocks in reverse order.
35 //!
36 //! Reads data records from the stream of blocks in reverse order.
37 //! \remark Reading performed in the background, i.e. with overlapping of I/O and computation
38 template <typename BlockType, typename BIDIteratorType>
39 class buf_istream_reverse : private noncopyable

67  //! Constructs input stream object, reading [first,last) blocks in reverse.
68  //! \param begin \c bid_iterator pointing to the first block of the stream
69  //! \param end \c bid_iterator pointing to the ( \b last + 1 ) block of the stream
70  //! \param nbuffers number of buffers for internal use
71  buf_istream_reverse(bid_iterator_type begin, bid_iterator_type end, int_type nbuffers)
72  : current_elem(0),
73 #ifdef BUF_ISTREAM_CHECK_END
74  not_finished(true),
75 #endif
76  bids_(end - begin)
77  {
78  // copy list of bids in reverse
79  std::reverse_copy(begin, end, bids_.begin());
80 
81  // calculate prefetch sequence
82  const unsigned_type ndisks = config::get_instance()->disks_number();
83  const unsigned_type mdevid = config::get_instance()->get_max_device_id();
84 
85  prefetch_seq = new int_type[bids_.size()];
86 
87  // optimal schedule
88  nbuffers = STXXL_MAX(2 * ndisks, unsigned_type(nbuffers - 1));
89  compute_prefetch_schedule(bids_.begin(), bids_.end(), prefetch_seq,
90  nbuffers, mdevid);
91 
92  // create stream prefetcher
93  prefetcher = new prefetcher_type(bids_.begin(), bids_.end(), prefetch_seq, nbuffers);
94 
95  // fetch block: last in sequence
96  current_blk = prefetcher->pull_block();
97  current_elem = block_type::size - 1;
98  }
99 
100  //! Input stream operator, reads in \c record.
101  //! \param record reference to the block record type,
102  //! contains value of the next record in the stream after the call of the operator
103  //! \return reference to itself (stream object)
104  _Self& operator >> (reference record)
105  {
106 #ifdef BUF_ISTREAM_CHECK_END
107  assert(not_finished);
108 #endif
109 
110  record = current_blk->elem[current_elem--];
111 
112  if (UNLIKELY(current_elem < 0))
113  {
114  current_elem = block_type::size - 1;
115 #ifdef BUF_ISTREAM_CHECK_END
116  not_finished = prefetcher->block_consumed(current_blk);
117 #else
118  prefetcher->block_consumed(current_blk);
119 #endif
120  }
121 
122  return (*this);
123  }
124 
125  //! Returns reference to the current record in the stream.
126  reference current() /* const */
127  {
128  return current_blk->elem[current_elem];
129  }
130 
131  //! Returns reference to the current record in the stream.
132  reference operator * () /* const */
133  {
134  return current_blk->elem[current_elem];
135  }
136 
137  //! Moves to the _previous_ record in the stream.
138  //! \return reference to itself after the advance
139  _Self& operator ++ ()
140  {
141 #ifdef BUF_ISTREAM_CHECK_END
142  assert(not_finished);
143 #endif
144 
145  current_elem--;
146 
147  if (UNLIKELY(current_elem < 0))
148  {
149  current_elem = block_type::size - 1;
150 #ifdef BUF_ISTREAM_CHECK_END
151  not_finished = prefetcher->block_consumed(current_blk);
152 #else
153  prefetcher->block_consumed(current_blk);
154 #endif
155  }
156  return *this;
157  }
158 
159  //! Frees used internal objects.
160  ~buf_istream_reverse()
161  {
162  delete prefetcher;
163  delete[] prefetch_seq;
164  }
165 };
